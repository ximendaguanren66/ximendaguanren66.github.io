搭建的apache + php + mysql的环境（要求使用php5.6）

＃安装httpd
yum install httpd httpd-devel httpd-manual mod_ssl -y
＃启动apache
systemctl start httpd
＃设置apache开机启动
systemctl enable httpd
＃查看开放端口，检查有没有httpd服务，80和443端口是否开放
ss -antlp
配置防火墙，可以正常访问阿帕奇

firewall-cmd –zone = public –add-port = 80 / tcp
–permanent firewall-cmd –zone = public –add-port = 433 / tcp
–permanent firewall-cmd –reload

安装php5.6
rpm -Uvh https://mirror.webtatic.com/yum/el7/epel-release.rpm
rpm -Uvh https://mirror.webtatic.com/yum/el7/webtatic-release.rpm#overload
system. php
yum删除php-common -y

＃安装php5.6和php扩展
yum install -y php56w php56w-opcache php56w-xml php56w-mcrypt php56w-gd php56w-mysql php56w-intl php56w-mbstring

＃进入网站根目录
cd / var / www / html /
＃在网站根目录中创建index.php，
写入phpinfo函数，测试网站php是否正常
echo ‘<？php phpinfo（）;？>’ > index.php

安装MariaDB的
yum install mariadb-server -y
＃启动mariadb-server
systemctl start mariadb
＃设置mariadb-server开机自启
systemctl enablemariadb
＃ mysql初始化，注意记住mysql密码
mysql_secure_installation
＃ 重启mariadb服务，读取my.cnf新配置
systemctl restart mariadb
＃ 登录数据库，查看字符编码
mysql -uroot -p
输入 \s 查看编码
